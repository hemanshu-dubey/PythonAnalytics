def cleanText(txt):
    txt= re.sub(r"http\S+", "", txt)
    txt=re.sub(r"(\\x(.){2})","",txt).lower()
    txt=re.sub(r"\W+"," ",txt).strip()
    txt=txt.replace(' rt ','')[2:]
    return txt.strip('\n')
